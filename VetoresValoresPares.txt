   val: vetor[1..7] de Inteiro
   i,TP: Inteiro

inicio
      tp <- 0
      Para i <- 1 ate 7 faca
           Escreva("Digite o ",i, "o. valor: ")
           Leia(val[i])

                Escreval("Valor PAR na Posicao: ", i)
